use vitcampus;
create table department (
did int primary key,
dname varchar(25) not null unique);
insert into department value (101,'cse'),(102,'ece'),(103,'civil'),(106,'aero'),(107,'mech');
select * from department order by did;
create table hod (
hid int primary key,
hname varchar(20) not null unique);
insert into hod values  (101,'arun '),(102,'jay'),(103,'sara'),(106,'clara');
select * from hod ;
SELECT * FROM  department CROSS JOIN hod order by did asc;
select * from department inner join hod where  department.did = hod.hid;
select * from department natural join hod ;
select * from department left outer join hod on (department.did=hod.hid) order by did asc;
select * from department full outer join  hod on (department.did=hod.hid);
